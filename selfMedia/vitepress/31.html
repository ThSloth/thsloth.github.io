安装vue，执行命令pnpm add vue

在之前创建的文件theme/index.ts，使用插槽，配置内容如下：

ts
import Theme from "vitepress/theme";
import { h } from "vue";
import AnimateTitle from "./components/AnimateTitle.vue";

export default {
  ...Theme,
  Layout() {
    return h(Theme.Layout, null, {
      "home-hero-info": () => h(AnimateTitle),
    });
  },
};
